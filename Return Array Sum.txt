import java.util.Scanner;
public class Solution {
	
	public static int sum(int[] arr) {
		//Your code goes here
        int sum=0;
        for(int x:arr){
            sum+=x;
        }
        return sum;
        
	}
    
    public static void main(String args[]){
        Scanner s=new Scanner(System.in);
        int num=s.nextInt();
        for (int i=1;i<=num;i++){
            int cases=s.nextInt();
            int arr[]=new int[cases];
            for (int j=0;j<cases;j++){
                arr[i]=s.nextInt();
                
            }
            
            sum(arr);
        }
    
    }
}
